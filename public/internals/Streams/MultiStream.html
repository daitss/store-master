def get
  if @ungot
    @ungot = false
    return @last[0], @last[1]
  end

  scorecard = []

  @streams.each do |s|
    k, v = s.get
    scorecard.push [ s, k, v ] unless k.nil?
  end

  return if scorecard.empty?

  key  = scorecard.map{ |s, k, v|  k  }.sort[0]
  vals = @values_container.new

  scorecard.each do |s, k, v|
    if k == key
      vals << v
    else
      s.unget
    end
  end

  @last = [ key, vals ]
  return    key, vals
end
