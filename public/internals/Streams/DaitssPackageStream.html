def get
  return if eos?

  if @ungot
     @ungot = false
     return @unget.pop, @unget.pop
  end

  if @buff.empty?
    @buff = DaitssModel::Package.package_copies @ieids.shift(2000)
  end

  rec = @buff.shift

  @unget = [ rec, rec.url ]

  return rec.url, rec
end
