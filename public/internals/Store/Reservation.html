def create_new_name ieid
    vers = '.000'
    while vers <= '.999' do
      res  = DM::Reservation.create(:name => ieid + vers, :ieid => ieid)
      return res.name if res.saved?
      vers.succ!
    end
    raise Store::DatabaseError, "Can't create a new name for IEID '#{ieid}'."
  end
